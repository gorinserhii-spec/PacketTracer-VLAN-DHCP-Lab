  enable
  conf t
  hostname Switch1

! Creating a VLAN
  vlan 10
  name VLAN10
  exit
  vlan 20
  name VLAN20
  exit

! Configuring Port for PC1 (VLAN 10)
  int fa0/1
  switchport mode access
  switchport access vlan 10

! Configuring port for PC2 (VLAN 20)
  int fa0/2
  switchport mode access
  switchport access vlan 20

! Configuring a Trunk Port for Router1
  int fa0/24
  switchport mode trunk
  switchport trunk allowed vlan 10,20
  end
