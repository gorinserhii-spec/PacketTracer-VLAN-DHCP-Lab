  enable
  conf t
  hostname Router2

! Настройка интерфейса, подключенного к Switch2
  int g0/0/0
  no ip add
  no shut

! Настройка подинтерфейса с нативной VLAN для Switch2
  int g0/0/0.10
  encapsulation dot1Q 10 native
  ip add 192.168.30.1 255.255.255.0
  no shut

! Настройка интерфейса, подключенного к Router1
  int g0/0/1
  ip add 10.0.0.2 255.255.255.252
  no shut

! Настройка DHCP-пула для PC3
  ip dhcp pool VLAN10_BRANCH
  network 192.168.30.0 255.255.255.0
  default-router 192.168.30.1

! Настройка статических маршрутов для сетей PC1 и PC2
  ip route 192.168.10.0 255.255.255.0 10.0.0.1
  ip route 192.168.20.0 255.255.255.0 10.0.0.1
  end
