  enable
  conf t
  hostname Switch1

! Создание VLAN
  vlan 10
  name VLAN10
  exit
  vlan 20
  name VLAN20
  exit

! Настройка порта для PC1 (VLAN 10)
  int fa0/1
  switchport mode access
  switchport access vlan 10

! Настройка порта для PC2 (VLAN 20)
  int fa0/2
  switchport mode access
  switchport access vlan 20

! Настройка транкового порта для Router1
  int fa0/24
  switchport mode trunk
  switchport trunk allowed vlan 10,20
  end
